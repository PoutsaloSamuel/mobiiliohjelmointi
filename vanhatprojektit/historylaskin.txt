import { StatusBar } from 'expo-status-bar';
import React from 'react';
import { Button, StyleSheet, Text, View, Alert, TextInput} from 'react-native';
import { NavigationContainer} from '@react-navigation/native';
import { createStackNavigator} from'@react-navigation/stack';
import { createBottomTabNavigator} from '@react-navigation/bottom-tabs';


const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#fff',
    alignItems: 'center',
    justifyContent: 'flex-end',
    padding: 20,
  },
  other: {
    flex: 1,
    backgroundColor: '#fff',
    alignItems: 'center',
    justifyContent: 'center',
    padding: 20,
  },
});


function HomeScreen({navigation}) {
  var [value1, setValue1] = React.useState('')
  var [value2, setValue2] = React.useState('')
  var [historia, setHistoria] = React.useState('')
  var [lasku, setLasku] = React.useState('')
  const v1 = parseInt(value1)
  const v2 = parseInt(value2)
  const Tab = createBottomTabNavigator();
  const plusLasku = () => {
    setLasku(v1 + v2)
    setHistoria(historia + v1 +  " + " + v2 + " = " + (v1 + v2) + '\n')
  };

  const miinusLasku = () => {
    setLasku(v1 - v2)
    setHistoria(historia + v1 +  " - " + v2 + " = " + (v1 - v2) + '\n')
  };
  

  return (
    <View style={styles.container}>
    <View style={styles.container}>
      <Text>Result: {lasku}</Text>
      <TextInput 
      style={{ height: 40, width: 200, borderColor: 'gray', borderWidth: 1 }}
      keyboardType = 'numeric' 
      onChangeText={value1 => setValue1(value1)}
      value={value1}/>
      <TextInput 
      style={{ height: 40, width: 200, borderColor: 'gray', borderWidth: 1 }} 
      keyboardType = 'numeric'
      onChangeText={value2 => setValue2(value2)}
      value={value2}/>
      <StatusBar style="auto" />
    </View>
    <View style={{flex: 1, flexDirection:'row', alignItems: 'flex-start', justifyContent: 'space-evenly', margin: 10}}>
    <Button onPress={plusLasku} title=" + " style={{}}/>
    <Button onPress={miinusLasku} title=" - " style={{}}/>
    <Button title="Settings" onPress={() => navigation.navigate('Settings', {user: historia})}/>
    
    </View>
    <View style={{flex: 2, flexDirection:'row', alignItems: 'flex-start', justifyContent: 'space-evenly'}}>
    
    </View>
    </View>

);
}


function SettingScreen({route}) {
  const {user} = route.params
  return (
    <View style={styles.other}>
      <Text>History{"\n"} {user}</Text>
    </View>
  
  );
  }

export default function App( {navigation}) {
  const Tab = createBottomTabNavigator();
  return (
    <View style={{flex: 1}}>
    <NavigationContainer>
    <Tab.Navigator>
      <Tab.Screen name="Home" component={HomeScreen} />
      <Tab.Screen name="Settings" component={SettingScreen} />
    </Tab.Navigator>
    </NavigationContainer>
    </View>
  );
}
