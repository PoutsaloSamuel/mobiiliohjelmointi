import { StatusBar } from 'expo-status-bar';
import React from 'react';
import { Button, StyleSheet, Text, View, Alert, TextInput} from 'react-native';

export default function App() {
  var [value, setValue] = React.useState('')
  var [vastaus, setVastaus] = React.useState('Guess a number between 1-100')
  var v1 = parseInt(value)
  var arvauksia = 0
  var numero = Math.floor(Math.random() * 100) + 1
  const arvaus = () => {
    numero = Math.floor(Math.random() * 100) + 1
    arvauksia++
    if (v1 == numero) {
      Alert.alert("You guessed the number in " + arvauksia + " guesses")
    }
    
    if (v1 > numero){
      setVastaus("Your guess " + v1 + " is high")
    }
    else {
      setVastaus("Your guess " + v1 + " is low")
    }
  };
  

  return (
    <View style={{flex: 1}}>
    <View style={styles.container}>
      <Text>{vastaus}</Text>
      <TextInput 
      style={{ height: 40, width: 200, borderColor: 'gray', borderWidth: 1 }}
      keyboardType = 'numeric' 
      onChangeText={value => setValue(value)}
      value={value}/>
      <StatusBar style="auto" />
    </View>
    <View style={{flex: 1, flexDirection:'row', alignItems: 'flex-start', justifyContent: 'space-evenly', margin: 10}}>
    <Button onPress={arvaus} title="MAKE GUESS" style={{}}/> 
    </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1.5,
    backgroundColor: '#fff',
    alignItems: 'center',
    justifyContent: 'flex-end',
    padding: 20,
  },
});
